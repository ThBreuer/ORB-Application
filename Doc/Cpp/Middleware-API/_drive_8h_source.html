165 virtual void fahre( int li, int re )
166 {
167 mLi.setSpeed( 0 );
168 mRe.setSpeed( 0 );
169
170 orb.wait( 200 );
171
172 int targetLi = mLi.getPosition() + li;
173 int targetRe = mRe.getPosition() + re;
174
175 // Passe Geschwindigkeit der beiden Raeder so an,
176 // dass diese gleichzeitig ihr Ziel erreichen
177 float dges = (fabs((float)li) + fabs((float)re));
178
179 mLi.setPosition( vmax * (float)li/dges, targetLi );
180 mRe.setPosition( vmax * (float)re/dges, targetRe );
181
182 orb.wait( 200 );
183
184 while( fabs( (float)mLi.getPosition() - targetLi ) > 1
185 || fabs( (float)mRe.getPosition() - targetRe ) > 1 )
186 {
187 }
188
189 orb.wait( 200 );
190 }
